from tacoma.epidemics import simulate_and_measure_i_inf

N = 100
k = 10
omega = 1.6
recovery_rate = 0.1
R0 = 10
t_run_total = 1000

AM = tc.EdgeActivityModel(N,
                       k/(N-1.),
                       omega,
                       t0 = 2000
                      )
infection_rate = R0 / k * recovery_rate


SIS = tc.SIS(N,t_run_total,infection_rate,recovery_rate,
        number_of_initially_infected=N,
        sampling_dt=0.0,
        )
